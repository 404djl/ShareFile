### [投资最大收益周期](https://codefun2000.com/p/P4552)

> 团团过年收获了很多压岁钱，妈妈帮他开了账户去投资。现在给出 n*n* 天内投资收益情况，选出划中连续多少天的收益总和量大，这个收益是多少。
>
> 第一行是一个整数 n ，表示天数，n 的范围为 [0,1000][0,1000]
>
> 第二行是 n 整数组成的一个数组，表示每天的收益，有正数也有负数，范围为 [−10000,100000]
>
> 输出一个整数，表示选取的连续天数的收益

```c++
#include <iostream>
#include <vector>
using namespace std;

// 计算最大连续子数组和（最大连续收益）
long long maxProfit(const vector<long long>& a) {
    if (a.empty()) return 0;

    long long cur = a[0];   // 以当前天结尾的最大和
    long long best = a[0];  // 全局最大和

    for (int i = 1; i < (int)a.size(); i++) {
        long long x = a[i];
        // 关键：要么从当前天重新开始，要么接到前面连续段后面
        cur = max(x, cur + x);
        best = max(best, cur);
    }
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<long long> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    cout << maxProfit(a) << "\n";
    return 0;
}
```

### [数据匹配](https://codefun2000.com/p/P4520)

> 输入一组以首个字符串为目标串、后续为待匹配串的字符串序列，筛选**从首字符开始完全前缀匹配**的待匹配串，按匹配长度从大到小排序，输出每个匹配字符串及其出现次数；无匹配结果则输出null。

```c++
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    vector<string> input;
    string s;

    while (cin >> s) {
        input.push_back(s);
    }

    if (input.empty()) return 0;

    string target = input[0];

    unordered_map<string, int> cnt;

    for (int i = 1; i < input.size(); i++) {
        string cur = input[i];

        // ❗长度限制 + 严格前缀匹配
        if (cur.size() <= target.size() &&
            target.substr(0, cur.size()) == cur) {
            cnt[cur]++;
        }
    }

    if (cnt.empty()) {
        cout << "null\n";
        return 0;
    }

    vector<pair<string, int>> vec(cnt.begin(), cnt.end());

    sort(vec.begin(), vec.end(), [&](auto &a, auto &b) {
        // 按匹配长度（字符串长度）降序
        if (a.first.size() != b.first.size())
            return a.first.size() > b.first.size();
        // 字典序
        return a.first < b.first;
    });

    for (auto &p : vec) {
        cout << p.first << " " << p.second << "\n";
    }

    return 0;
}
```

### [奶茶店的超级收益日](https://codefun2000.com/p/P4524)

> 商场新开了一家奶茶店，店长 A*A* 每天统计当天的收入，连续统计 *n* 天，用以查看收入的趋势。如果某一天的收入比后面某一天的两倍收入还多，则认为这一天是超级收益日。要求计算这段时间内这样的超圾收益日有多少天。

```c++
#include <bits/stdc++.h>
using namespace std;

// 功能函数：统计超级收益日的天数
int countSuperDays(const vector<int> &a) {
    int n = (int)a.size();
    if (n <= 1) {
        // 不足两天，没有超级收益日
        return 0;
    }

    int suffixMin = a[n - 1]; // 后缀最小值，初始为最后一天
    int count = 0;

    // 从倒数第二天开始往前遍历
    for (int i = n - 2; i >= 0; --i) {
        // 判断当前天是否为超级收益日
        if (a[i] > 2 * suffixMin) {
            count++;
        }
        // 更新后缀最小值
        if (a[i] < suffixMin) {
            suffixMin = a[i];
        }
    }

    return count;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    // 读取天数
    if (!(cin >> n)) {
        return 0;
    }

    // 读取每天的收入
    vector<int> incomes(n);
    for (int i = 0; i < n; ++i) {
        cin >> incomes[i];
    }

    // 调用功能函数并输出结果
    int ans = countSuperDays(incomes);
    cout << ans << "\n";

    return 0;
}
```

### [求一组算子的最短执行时间](https://codefun2000.com/p/P2271)

> 给定 n 个矢量算子，每个算子可选择放**矩阵单元**（耗时 nums [i]）或**向量单元**（耗时 6×nums [i]）；要求放在向量单元的算子必须是**原顺序连续一段**，其余放矩阵单元；总耗时为矩阵单元总耗时、向量单元总耗时二者的最大值，求该最大值的最小可能值。

```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int MAX_SIZE = 10010;

int main() {
    // 算子数量
    int n;
    cin >> n;

    // 存储每个算子的执行时间 + 前缀和数组
    vector<int> exec_time(MAX_SIZE);
    vector<int> prefix_sum(MAX_SIZE, 0);

    // 输入数据并计算前缀和
    for (int i = 1; i <= n; ++i) {
        cin >> exec_time[i];
        prefix_sum[i] = prefix_sum[i - 1] + exec_time[i];
    }

    // 总执行时间（全部算子在矩阵单元）
    int total_matrix = prefix_sum[n];
    // 初始化最优解：全部算子在矩阵单元
    int min_total_time = total_matrix;

    // 枚举所有可分配到向量单元的区间 [i,j]
    for (int i = 1; i <= n; ++i) {
        for (int j = i; j <= n; ++j) {
            // 区间 [i,j] 的总执行时间
            int vec_segment = prefix_sum[j] - prefix_sum[i - 1];
            
            // 剪枝：当前区间已无法更优，直接跳出内层循环
            if (vec_segment * 6 >= min_total_time) {
                break;
            }

            // 向量单元耗时 *6，矩阵单元耗时 = 总耗时 - 区间耗时
            int vec_time = vec_segment * 6;
            int matrix_time = total_matrix - vec_segment;
            
            // 最终耗时取两者最大值
            int current_max = max(vec_time, matrix_time);
            // 更新最优解
            min_total_time = min(min_total_time, current_max);
        }
    }

    // 输出最小总执行时间
    cout << min_total_time << endl;

    return 0;
}
```



### [海蒂爷爷的木桶](https://codefun2000.com/p/P3793)

> 给定一个满足**循环单调非递减**的环形木板数组，插入一块新木板，找到**最靠左**的合法插入位置，保证插入后仍满足循环非递减规律，输出插入后的完整数组。

```c++
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;

    vector<int> boards(n);
    for (int i = 0; i < n; i++) {
        cin >> boards[i];
    }

    int x;
    cin >> x;

    int pos = -1;

    for (int i = 0; i < n; i++) {
        int a = boards[i];
        int b = boards[(i + 1) % n];

        // 情况1：正常递增区间
        if (a <= x && x <= b) {
            pos = i + 1;
            break;
        }

        // 情况2：断点
        if (a > b) {
            if (x >= a || x <= b) {
                pos = i + 1;
                break;
            }
        }
    }

    // 如果没找到（全部一样等情况）
    if (pos == -1) pos = 0;

    // 插入
    boards.insert(boards.begin() + pos, x);

    // 输出
    for (int i = 0; i < boards.size(); i++) {
        cout << boards[i];
        if (i != boards.size() - 1) cout << " ";
    }

    return 0;
}
```



## BFS

### *[关灯](https://codefun2000.com/p/P2302)

> 小明需从屋子内标记为 “s” 的位置出发，通过上下左右移动到达标记为 “t” 的电源处，初始有一定水深且水面每单位时间上涨 1，小明每移动一步耗时 1 单位时间，仅能移动到未被水淹的方格（箱子方格需高度大于当前水位，0 方格必被淹，s 和 t 始终安全），若无法找到此类路线，小明需待在原地，最终需输出路径长度及各步坐标。

```c++
#include <bits/stdc++.h>
using namespace std;

typedef pair<int,int> P;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H;
    cin >> H;

    int n, m;
    cin >> n >> m;

    vector<vector<string>> grid(n, vector<string>(m));
    vector<vector<int>> height(n, vector<int>(m, 0));

    P start, target;

    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            cin >> grid[i][j];
            if(grid[i][j] == "s") {
                start = {i, j};
                height[i][j] = 1e9; // 永不淹
            }
            else if(grid[i][j] == "t") {
                target = {i, j};
                height[i][j] = 1e9;
            }
            else {
                int h = stoi(grid[i][j]);
                height[i][j] = h;
            }
        }
    }

    // 最晚进入时间
    vector<vector<int>> deadline(n, vector<int>(m));
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            deadline[i][j] = height[i][j] - H;
        }
    }

    // BFS
    queue<P> q;
    vector<vector<int>> dist(n, vector<int>(m, -1));
    vector<vector<P>> pre(n, vector<P>(m, {-1,-1}));

    q.push(start);
    dist[start.first][start.second] = 0;

    int dx[4] = {1, -1, 0, 0};
    int dy[4] = {0, 0, 1, -1};

    while(!q.empty()) {
        auto [x, y] = q.front(); q.pop();

        for(int k = 0; k < 4; k++) {
            int nx = x + dx[k];
            int ny = y + dy[k];

            if(nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
            if(dist[nx][ny] != -1) continue;

            int nt = dist[x][y] + 1;

            // 必须在淹没前进入
            if(nt >= deadline[nx][ny]) continue;

            dist[nx][ny] = nt;
            pre[nx][ny] = {x, y};
            q.push({nx, ny});
        }
    }

    // 无法到达
    if(dist[target.first][target.second] == -1) {
        cout << 1 << "\n";
        cout << start.first << " " << start.second << "\n";
        return 0;
    }

    // 还原路径
    vector<P> path;
    P cur = target;
    while(cur != make_pair(-1, -1)) {
        path.push_back(cur);
        cur = pre[cur.first][cur.second];
    }

    reverse(path.begin(), path.end());

    cout << path.size() << "\n";
    for(auto &p : path) {
        cout << p.first << " " << p.second << "\n";
    }

    return 0;
}
```
