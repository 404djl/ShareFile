### [投资最大收益周期](https://codefun2000.com/p/P4552)

> 团团过年收获了很多压岁钱，妈妈帮他开了账户去投资。现在给出 n*n* 天内投资收益情况，选出划中连续多少天的收益总和量大，这个收益是多少。
>
> 第一行是一个整数 n ，表示天数，n 的范围为 [0,1000][0,1000]
>
> 第二行是 n 整数组成的一个数组，表示每天的收益，有正数也有负数，范围为 [−10000,100000]
>
> 输出一个整数，表示选取的连续天数的收益

```c++
#include <iostream>
#include <vector>
using namespace std;

// 计算最大连续子数组和（最大连续收益）
long long maxProfit(const vector<long long>& a) {
    if (a.empty()) return 0;

    long long cur = a[0];   // 以当前天结尾的最大和
    long long best = a[0];  // 全局最大和

    for (int i = 1; i < (int)a.size(); i++) {
        long long x = a[i];
        // 关键：要么从当前天重新开始，要么接到前面连续段后面
        cur = max(x, cur + x);
        best = max(best, cur);
    }
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<long long> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    cout << maxProfit(a) << "\n";
    return 0;
}
```

### [无线网络覆盖计划](https://codefun2000.com/p/P3306)

这是一个典型的 0-1 背包问题。我们可以把总预算看作背包的容量，每个无线接入点的安装成本看作物品的重量，覆盖范围看作物品的价值。题目要求在满足覆盖需求的前提下，使得总成本最小；如果成本相同，则选择覆盖范围最大的方案。

由于题目明确说明预算 `budget` 和安装成本 `cost` 都是 10 的整数倍，我们可以将成本除以 10 进行状态压缩，从而大幅度降低空间和时间复杂度。

**算法思路**

1. **状态定义**：设 `dp[j]` 表示在花费预算为 `j`（实际预算为 `j * 10`）时，能够获得的最大覆盖面积。
2. **状态初始化**：`dp[0] = 0`，表示花费为 0 时覆盖面积为 0；其余 `dp[j]` 初始化为 `-1`，表示该花费不可达。
3. **状态转移**：对于每一个接入点（覆盖面积为 `c`，成本为 `v`），逆序遍历预算数组，若前置状态 `dp[j - v/10]` 可达，则更新 `dp[j] = max(dp[j], dp[j - v/10] + c)`。
4. **结果查找**：遍历 `dp` 数组，从小到大寻找第一个满足 `dp[j] >= areaRequirement` 的 `j`，此即为满足条件的最小预算，对应的 `dp[j]` 即为最大覆盖面积；若找不到，则输出 `0 0`。

**C++ 代码实现**

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    // 优化输入输出流速度
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long areaRequirement;
    int budget, n;
    if (!(cin >> areaRequirement >> budget >> n)) return 0;

    // 成本与预算都是10的倍数，压缩状态空间
    int max_b = budget / 10;
    vector<long long> dp(max_b + 1, -1);
    dp[0] = 0;

    for (int i = 0; i < n; ++i) {
        long long coverage;
        int cost;
        cin >> coverage >> cost;
        
        int cc = cost / 10;
        if (cc > max_b) continue; // 剪枝：超出总预算的接入点直接忽略
        
        // 0-1背包逆序遍历
        for (int j = max_b; j >= cc; --j) {
            if (dp[j - cc] != -1) {
                dp[j] = max(dp[j], dp[j - cc] + coverage);
            }
        }
    }

    int min_cost = -1;
    long long max_cov = 0;
    
    // 从小到大寻找满足覆盖需求的最小预算
    for (int j = 0; j <= max_b; ++j) {
        if (dp[j] >= areaRequirement) {
            min_cost = j * 10; // 还原真实预算
            max_cov = dp[j];
            break;
        }
    }

    if (min_cost == -1) {
        cout << 0 << " " << 0 << "\n";
    } else {
        cout << min_cost << " " << max_cov << "\n";
    }

    return 0;
}
```

### *[服务器休息计划](https://codefun2000.com/p/P2607)

> 高速公路有**N 个服务区**，连续经过**最多 M 个服务区**就必须休息一次，选择若干服务区休息并花费对应费用，求满足规则下**休息总花费最小**的方案。

```c++
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 10005;
int N, M;
int cost[MAXN];
int dp[MAXN];

int main() {
    cin >> N >> M;
    for (int i = 1; i <= N; i++) {
        cin >> cost[i];
    }

    deque<int> q; // 存下标，维护 dp 单调递增

    for (int i = 1; i <= N; i++) {
        // 移除窗口外元素
        while (!q.empty() && q.front() < i - M) {
            q.pop_front();
        }

        if (i <= M) {
            dp[i] = cost[i];
        } else {
            dp[i] = cost[i] + dp[q.front()];
        }

        // 维护单调性
        while (!q.empty() && dp[q.back()] >= dp[i]) {
            q.pop_back();
        }
        q.push_back(i);
    }

    int ans = INT_MAX;
    for (int i = max(1, N - M + 1); i <= N; i++) {
        ans = min(ans, dp[i]);
    }

    cout << ans << endl;
    return 0;
}
```

### [D路通信](https://codefun2000.com/p/P2404)

> 给定两个仅含小写字母的字符串，求它们**最长连续公共子串**的长度，子串可以为空，只需输出该长度数值。

```c++
#include <bits/stdc++.h>
using namespace std;

int main() {
    string s1, s2;
    cin >> s1 >> s2;

    int n = s1.size();
    int m = s2.size();

    vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
    int ans = 0;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            if (s1[i - 1] == s2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
                ans = max(ans, dp[i][j]);
            } else {
                dp[i][j] = 0;
            }
        }
    }

    cout << ans << endl;
    return 0;
}
```

### *[奖品兑换组合](https://codefun2000.com/p/P4234)

> 游乐场通过进行游戏获取游戏币，游戏币可以用来兑换奖品，每个奖品价值不同的游戏币数量。可兑换的奖品列表通过数组 values 给出，其中 values[i]表示兑换第 i个奖品价值的游戏币数量，**价值相同则记为同一奖品**。给出获取的游戏币数量 num，请计算刚好价值 num个游戏币的奖品组合的个数，如果不存在价值 num个游戏币的奖品组合，则返回 0 。

```c++
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<int> values(n);
    for (int i = 0; i < n; i++) cin >> values[i];

    int num;
    cin >> num;

    // 1️⃣ 按价值分组（value -> count）
    unordered_map<int, int> mp;
    for (int v : values) mp[v]++;

    // 2️⃣ 多重背包（组合计数）
    vector<long long> dp(num + 1, 0);
    dp[0] = 1;

    for (auto &it : mp) {
        int v = it.first;
        int cnt = it.second;

        vector<long long> next(num + 1, 0);

        for (int s = 0; s <= num; s++) {
            if (dp[s] == 0) continue;

            // 枚举用 k 个该价值
            for (int k = 0; k <= cnt && s + k * v <= num; k++) {
                next[s + k * v] += dp[s];
            }
        }

        dp.swap(next);
    }

    cout << dp[num] << "\n";
    return 0;
}
```

### [游戏中的地图穿越](https://codefun2000.com/p/P2958)

> 给定 k 阶正方形矩阵，从左上角 (0,0) 出发，**只能向右、向下移动**，相邻位置高度差超过 1 则不可通行；经过每个位置消耗对应高度体力，求到达矩阵右侧任意位置的**最小总体力消耗**；无路返回 - 1，参数不合法返回 - 2。

```c++
#include <bits/stdc++.h>
using namespace std;

int main() {
    int k;
    if (!(cin >> k)) return 0;
    // 参数 k 合法性检查
    if (k <= 0 || k > 100) {
        cout << -2;
        return 0;
    }
    vector<vector<int>> mp(k, vector<int>(k));
    for (int i = 0; i < k; i++) {
        for (int j = 0; j < k; j++) {
            cin >> mp[i][j];
            // 地形高度合法性检查
            if (mp[i][j] < 0 || mp[i][j] > 10) {
                cout << -2;
                return 0;
            }
        }
    }
    const int INF = 1e9;
    vector<vector<int>> dp(k, vector<int>(k, INF));
    dp[0][0] = mp[0][0];  // 起点消耗

    // 动态规划填表
    for (int i = 0; i < k; i++) {
        for (int j = 0; j < k; j++) {
            if (i > 0 && abs(mp[i][j] - mp[i-1][j]) <= 1) {
                dp[i][j] = min(dp[i][j], dp[i-1][j] + mp[i][j]);
            }
            if (j > 0 && abs(mp[i][j] - mp[i][j-1]) <= 1) {
                dp[i][j] = min(dp[i][j], dp[i][j-1] + mp[i][j]);
            }
        }
    }

    // 从最右列任意位置出去
    int ans = INF;
    for (int i = 0; i < k; i++) {
        ans = min(ans, dp[i][k-1]);
    }
    // 输出结果
    if (ans == INF) cout << -1;
    else cout << ans;
    return 0;
}
```

### *[网络整改](https://codefun2000.com/p/P3287)

> 给定以**1 为根**的无根树，先构建有根树并找出所有叶子边缘节点；删除最少数量节点，使剩余所有叶子节点到根的深度完全相等，求最小删除数量。

```cpp
#include <bits/stdc++.h>
using namespace std;

const int INF = 1e9;

int n;
vector<vector<int>> adj;
vector<int> depth;
vector<vector<int>> children;
int maxDepth;

// 计算深度+构建子树结构
void dfsDepth(int u, int p) {
    for (int v : adj[u]) {
        if (v == p) continue;
        depth[v] = depth[u] + 1;
        maxDepth = max(maxDepth, depth[v]);
        children[u].push_back(v);
        dfsDepth(v, u);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n;
    adj.assign(n+1, {});
    for (int i = 0; i < n-1; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    depth.assign(n+1, 0);
    children.assign(n+1, {});
    maxDepth = 0;
    dfsDepth(1, 0);

    vector<int> best(n+1);
    int max_save = 0;

    // 枚举所有可能的 目标叶子深度 h
    for (int h = 0; h <= maxDepth; h++) {
        // 按深度分层，自底向上DP
        vector<vector<int>> byDepth(maxDepth+1);
        for (int v = 1; v <= n; v++) {
            byDepth[depth[v]].push_back(v);
        }
        
        fill(best.begin(), best.end(), -INF);
        
        // 从最深的节点往根节点计算
        for (int d = maxDepth; d >= 0; d--) {
            for (int v : byDepth[d]) {
                if (depth[v] > h) {
                    // 深度超过目标，无法保留
                    best[v] = -INF;
                } 
                else if (depth[v] == h) {
                    // 刚好是目标深度，作为叶子，保留1个节点
                    best[v] = 1;
                } 
                else {
                    // 累加所有合法子树的节点数
                    int sum = 0;
                    for (int u : children[v]) {
                        // 修复点：只要子树合法（不是-INF）就累加
                        if (best[u] != -INF) {
                            sum += best[u];
                        }
                    }
                    if (sum == 0) {
                        best[v] = -INF;
                    } else {
                        best[v] = sum + 1; // +1 是当前节点自身
                    }
                }
            }
        }
        // 更新最大保留节点数
        max_save = max(max_save, best[1]);
    }

    // 答案 = 总设备数 - 最大保留数
    cout << n - max_save << endl;
    return 0;
}
```

### [命令行参数提示](https://codefun2000.com/p/P3291)

>本题要求在大量已知子命令中，根据用户输入的错误命令，找到与之最相似（即编辑距离最小）的子命令并提示。 最相似的定义为**Levenshtein 距离**，即将一个字符串通过 **替换**、**插入**、**删除** 三种基本操作变为另一个字符串所需的最少步数。

```CPP
#include <bits/stdc++.h>
using namespace std;

// 计算编辑距离
int editDist(const string &a, const string &b) {
    int n = a.size(), m = b.size();
    vector<vector<int>> dp(n+1, vector<int>(m+1));
    for (int i = 0; i <= n; i++) dp[i][0] = i;
    for (int j = 0; j <= m; j++) dp[0][j] = j;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            if (a[i-1] == b[j-1]) dp[i][j] = dp[i-1][j-1];
            else dp[i][j] = min({dp[i-1][j]+1, dp[i][j-1]+1, dp[i-1][j-1]+1});
        }
    }
    return dp[n][m];
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, N;
    cin >> D >> N;
    vector<string> cmds(N);
    for (int i = 0; i < N; i++) cin >> cmds[i];
    string inp;
    cin >> inp;

    // 精确匹配
    for (auto &cmd : cmds) {
        if (cmd == inp) {
            cout << inp << "\n";
            return 0;
        }
    }

    // 筛选候选
    vector<pair<int,string>> cand;
    for (auto &cmd : cmds) {
        int d = editDist(inp, cmd);
        if (d >= 1 && d <= D) {
            cand.emplace_back(d, cmd);
        }
    }

    if (cand.empty()) {
        cout << "None\n";
    } else {
        sort(cand.begin(), cand.end(), [](auto &a, auto &b) {
            if (a.first != b.first) return a.first < b.first;
            return a.second < b.second;
        });
        bool first = true;
        for (auto &p : cand) {
            if (!first) cout << ' ';
            cout << p.second;
            first = false;
        }
        cout << '\n';
    }
    return 0;
}
```

### [滑雪冒险](https://codefun2000.com/p/P2351)

> 给定 R 行 C 列高度矩阵，每次只能向**上下左右相邻**位置滑动，且下一点高度**严格更小**，求包含节点数最多的最长滑雪路径长度。



```cpp
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

const int MAX = 205;
int grid[MAX][MAX];   // 存储高度
int dp[MAX][MAX];     // 记忆化数组：dp[i][j] 表示从 (i,j) 出发的最长滑雪路径长度
int R, C;             // 行、列

// 四个方向：上下左右
int dirs[4][2] = {{-1,0}, {1,0}, {0,-1}, {0,1}};

// DFS + 记忆化搜索
int dfs(int x, int y) {
    // 已经计算过，直接返回结果（记忆化核心）
    if (dp[x][y] != 0) return dp[x][y];
    
    int max_len = 1; // 至少包含自己，长度为 1
    
    // 遍历四个方向
    for (int i = 0; i < 4; i++) {
        int nx = x + dirs[i][0];
        int ny = y + dirs[i][1];
        
        // 判断是否在边界内，且下一个点高度更低
        if (nx >= 0 && nx < R && ny >= 0 && ny < C && grid[nx][ny] < grid[x][y]) {
            max_len = max(max_len, dfs(nx, ny) + 1);
        }
    }
    
    dp[x][y] = max_len; // 保存结果
    return max_len;
}

int main() {
    cin >> R >> C;
    
    // 输入高度
    for (int i = 0; i < R; i++) {
        for (int j = 0; j < C; j++) {
            cin >> grid[i][j];
        }
    }
    
    memset(dp, 0, sizeof(dp)); // 初始化记忆数组为 0
    int ans = 0;
    
    // 遍历每个点作为起点
    for (int i = 0; i < R; i++) {
        for (int j = 0; j < C; j++) {
            ans = max(ans, dfs(i, j));
        }
    }
    
    cout << ans << endl;
    return 0;
}
```

### [圣诞节礼盒](https://codefun2000.com/p/P2307)

> 给定 N 个长、宽、高各不相同的礼盒，堆叠时**下方盒子长宽高必须严格大于上方盒子**，可任意旋转礼盒调整长宽高顺序，求堆叠礼盒高度总和的最大值。

```cpp
#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e3 + 10;
int dp[maxn];
int n;
struct node {
    int l, w, h;
}box[maxn];

int main() {
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        cin >> box[i].l >> box[i].w >> box[i].h;
    }

    sort(box + 1, box + n + 1, [&](node a, node b){
        if (a.l != b.l) {
            return a.l < b.l;
        }
        if (a.w != b.w) {
            return a.w < b.w;
        }
        return a.h < b.h;
    });

    int ans = 0;

    for (int i = 1; i <= n; i++) {
        dp[i] = box[i].h;
        for (int j = 1; j < i; j++) {
            if (box[i].h > box[j].h && box[i].l > box[j].l && box[i].w > box[j].w) {
                dp[i] = max(dp[i], dp[j] + box[i].h);
            }
        }
        ans = max(ans, dp[i]);
    }

    cout << ans << endl;
    return 0;

}
```

### *[快乐校园跑](https://codefun2000.com/p/P2388)

> 给定**有向带权图**（n 个建筑节点、m 条单向道路），节点自身权值是小游戏耗时、边权是道路跑步耗时；小队可多人分头走不同路径，从起点 x 出发，要求**优先经过必经点数量最多**，在此前提下总耗时取小队里所有人耗时的最大值最小；输出最多必经点数量、以及对应的最终成绩时间

```cpp
#include <bits/stdc++.h>
using namespace std;

using ll = long long;
const ll INF = (ll)4e18;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    // f[i][j]：从 i 到 j 的最短时间（初始化为无穷大）
    vector<vector<ll>> f(n + 1, vector<ll>(n + 1, INF));

    // a[i]：节点 i 的“自耗时”（到达该节点后额外消耗）
    vector<ll> a(n + 1, 0);

    // 读入边信息
    for (int i = 0, u, v; i < m; i++) {
        ll r;
        cin >> u >> v >> r;

        // 如果是自环，表示该点自身耗时
        if (u == v) a[u] = r;

        // 否则是普通有向边
        else f[u][v] = min(f[u][v], r);
    }

    // 将“点权 a[j]”加到所有 i -> j 的路径上
    // 表示：到达 j 必须额外耗时 a[j]
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= n; j++)
            if (i != j && f[i][j] < INF)
                f[i][j] += a[j];

    // 自己到自己距离初始化为 0
    for (int i = 1; i <= n; i++)
        f[i][i] = 0;

    // Floyd-Warshall 三重循环
    for (int k = 1; k <= n; k++)          // 中转点
        for (int i = 1; i <= n; i++)      // 起点
            for (int j = 1; j <= n; j++) {// 终点

                if (i == j) continue;

                // 如果 i -> k 和 k -> j 都可达
                if (f[i][k] < INF && f[k][j] < INF) {

                    // 两种情况：
                    // 1) 正常路径拼接 i->k + k->j
                    // 2) 如果 j == k，需要额外加 k 的点权 a[k]
                    f[i][j] = min(
                        f[i][j],
                        k == j ? f[i][k] + a[k]
                               : f[i][k] + f[k][j]
                    );
                }
            }

    int s;
    cin >> s;

    ll cnt = 1;   // 可达节点数（包含起点）
    ll ans = 0;   // 所有可达点中的最大耗时

    // 统计从 s 出发的结果
    for (int i = 1; i <= n; i++) {
        if (i == s) continue;

        // 如果可达
        if (f[s][i] < INF) {
            cnt++;                 // 可达点数量 +1
            ans = max(ans, f[s][i]); // 更新最坏时间（最大路径耗时）
        }
    }

    cout << cnt << "\n" << ans << "\n";
}
```

### **[打怪升级](https://codefun2000.com/p/P4554)

```c++
struct Monster {

pair<int,int> run(int lam) {
    vector<int> dpAdj(E + 1, NEG); // adj = sumY - lam * count
    vector<int> dpY(E + 1, 0);
    vector<int> dpC(E + 1, 0);

    for (auto &m : a) {
        for (int e = E - m.z; e >= 0; e--) {
            if (dpAdj[e] == NEG) continue;

            // 经验不够
            if (W + dpY[e] < m.x) continue;

            int ne = e + m.z;
            int nadj = dpAdj[e] + m.y - lam;
            int ny = dpY[e] + m.y;
            int nc = dpC[e] + 1;

            // 比较优劣
            if (nadj > dpAdj[ne] ||
               (nadj == dpAdj[ne] && ny > dpY[ne]) ||
               (nadj == dpAdj[ne] && ny == dpY[ne] && nc > dpC[ne])) {

                dpAdj[ne] = nadj;
                dpY[ne] = ny;
                dpC[ne] = nc;
            }
        }
    }

    int bestAdj = NEG, bestY = 0, bestC = 0;

    for (int e = 0; e <= E; e++) {
        if (dpAdj[e] == NEG) continue;

        if (dpAdj[e] > bestAdj ||
           (dpAdj[e] == bestAdj && dpY[e] > bestY) ||
           (dpAdj[e] == bestAdj && dpY[e] == bestY && dpC[e] > bestC)) {

            bestAdj = dpAdj[e];
            bestY = dpY[e];
            bestC = dpC[e];
        }
    }

        if (cnt >= K) l = mid;
        else r = mid - 1;

    for (auto &m : a) {

            if (W + dpY[e] < m.x) continue;

            int nadj = dpAdj[e] + m.y - lam;
            int ny = dpY[e] + m.y;
            int nc = dpC[e] + 1;

            if (nadj > dpAdj[ne] ||
               (nadj == dpAdj[ne] && ny > dpY[ne]) ||

                dpAdj[ne] = nadj;
                dpY[ne] = ny;
                dpC[ne] = nc;
            }
        }
    }

    int bestVal = NEG;

        int val = dpAdj[e] + lam * K;

        if (dpC[e] <= K) {

                bestY = dpY[e];
```

### *[日志文件存储问题](https://codefun2000.com/p/P2299)

> 共有**N 个进程**，每个进程有**M 个日志文件**，每个文件有大小和下载次数；U 盘总容量为 C，要求**每个进程必须至少选 1 个日志存入 U 盘**，在总占用容量不超过 C 的前提下，使得选中文件的下载次数总和最大，无解返回 - 1。

```cpp
#include <vector>
#include <algorithm>

    int n, m, c;
    cin >> n >> m >> c;

        }

    // 先对每个分组进行01背包
    unordered_map<int, vector<int>> dp;  // 使用unordered_map存储每个分组的dp结果

    for (auto &entry : a) {
        int idx = entry.first;

        for (auto &item : entry.second) {
            int sz = item.first, value = item.second;
            // 从后往前遍历，确保01背包的更新
            for (int i = c; i >= sz; --i) {
                if (dp[idx][i - sz] != -1) {
                    dp[idx][i] = max(dp[idx][i], dp[idx][i - sz] + value);
                }
            }
        }
    }

    // 01背包结束，开始把 dp 的结果当作物品继续进行01背包
    vector<int> f(c + 1, -1);
    f[0] = 0;  // 一开始f[0] 合法，是因为需要让第一个分组能够成功转移

    for (auto &entry : a) {
        int idx = entry.first;
        vector<int> tmp(c + 1, -1);

        // i 为总共大小
        for (int i = 0; i <= c; ++i) {
            // j是当前分组分配的大小
            for (int j = 1; j <= i; ++j) {
                if (f[i - j] != -1 && dp[idx][j] != -1) {
                    tmp[i] = max(tmp[i], f[i - j] + dp[idx][j]);
                }
            }
        }
        f = tmp;
    }

    cout << *max_element(f.begin(), f.end()) << endl;
```

### [求一组算子的最短执行时间](https://codefun2000.com/p/P2271)

> 给定 n 个矢量算子，每个算子可选择放**矩阵单元**（耗时 nums [i]）或**向量单元**（耗时 6×nums [i]）；要求放在向量单元的算子必须是**原顺序连续一段**，其余放矩阵单元；总耗时为矩阵单元总耗时、向量单元总耗时二者的最大值，求该最大值的最小可能值。

```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int MAX_SIZE = 10010;

int main() {
    // 算子数量
    int n;
    cin >> n;

    // 存储每个算子的执行时间 + 前缀和数组
    vector<int> exec_time(MAX_SIZE);
    vector<int> prefix_sum(MAX_SIZE, 0);

    // 输入数据并计算前缀和
    for (int i = 1; i <= n; ++i) {
        cin >> exec_time[i];
        prefix_sum[i] = prefix_sum[i - 1] + exec_time[i];
    }

    // 总执行时间（全部算子在矩阵单元）
    int total_matrix = prefix_sum[n];
    // 初始化最优解：全部算子在矩阵单元
    int min_total_time = total_matrix;

    // 枚举所有可分配到向量单元的区间 [i,j]
    for (int i = 1; i <= n; ++i) {
        for (int j = i; j <= n; ++j) {
            // 区间 [i,j] 的总执行时间
            int vec_segment = prefix_sum[j] - prefix_sum[i - 1];
            
            // 剪枝：当前区间已无法更优，直接跳出内层循环
            if (vec_segment * 6 >= min_total_time) {
                break;
            }

            // 向量单元耗时 *6，矩阵单元耗时 = 总耗时 - 区间耗时
            int vec_time = vec_segment * 6;
            int matrix_time = total_matrix - vec_segment;
            
            // 最终耗时取两者最大值
            int current_max = max(vec_time, matrix_time);
            // 更新最优解
            min_total_time = min(min_total_time, current_max);
        }
    }

    // 输出最小总执行时间
    cout << min_total_time << endl;

    return 0;
}
```

### *[求最大括号表达式](https://codefun2000.com/p/P2274)

> 给定一个**合法有效括号串**，可任意次数交换**相邻的两个独立有效子括号表达式**，将括号转为左 1 右 0 二进制串，求二进制数值**最大**的最终括号串。

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

// 自定义比较器函数，比较两个字符串拼接后的字典序
bool compare(const string& a, const string& b) {
    return a + b < b + a;
}

// 深度优先搜索函数，递归处理括号表达式
string dfs(string s) {
    if (s.empty()) return "";

    vector<string> part;
    string now;
    int cnt = 0;

    // 分割括号表达式，按完整的括号对划分
    for (char c : s) {
        if (c == '(') cnt++;
        else cnt--;
        now += c;
        if (cnt == 0) { // 找到一个完整的括号对
            part.push_back(now);
            now.clear();
        }
    }

    vector<string> subRes;

    if (part.size() == 1) {
        // 如果只有一个完整的括号对，递归处理去掉最外层括号后的子表达式
        subRes.push_back("(" + dfs(part[0].substr(1, part[0].size() - 2)) + ")");
    } else {
        // 递归处理多个括号对
        for (string p : part) {
            subRes.push_back(dfs(p));
        }
    }

    // 排序子表达式，使得字典序最大
    sort(subRes.begin(), subRes.end(), compare);

    // 返回拼接后的结果
    string result;
    for (const string& str : subRes) {
        result += str;
    }
    return result;
}

    cout << dfs(s) << endl; // 输出处理后的结果
    return 0;
```

```cpp
    vector<vector<long long>> pre(2, vector<long long>(m + 1, 0));

    // 输入 + 前缀和
    for (int i = 0; i < m; i++) {
        cin >> a[0][i];
        pre[0][i + 1] = pre[0][i] + a[0][i];
    }

    for (int i = 0; i < m; i++) {
        cin >> a[1][i];
        pre[1][i + 1] = pre[1][i] + a[1][i];
    }

    long long ans = LLONG_MAX;

    // 枚举小A的转折点 i
    for (int i = 0; i < m; i++) {

        // 小B可能吃的最大值（两种路径情况）
        long long b1 = pre[1][i]; // 下层左边
        long long b2 = pre[0][m] - pre[0][i + 1]; // 上层右边

        long long tmp = max(b1, b2);

        ans = min(ans, tmp);
    }

    cout << ans << "\n";

    return 0;
}
```

### [计算式](https://codefun2000.com/p/P2357)

> 给定两个多项式的系数，第i项表示指数为n-i的系数。输出两个多项式的求和结果。

```cpp
#include <bits/stdc++.h>
using namespace std;

// 加法运算
vector<int> add(vector<int>& a, vector<int>& b) {
    int n = max(a.size(), b.size());
    vector<int> res(n, 0);
    // 将两个数组逐位相加
    for (int i = 0; i < n; i++) {
        if (i < a.size()) res[i] += a[i];
        if (i < b.size()) res[i] += b[i];
    }
    reverse(res.begin(), res.end());
    // 去除前导零
    while (res.size() > 1 && res[0] == 0) {
        res.erase(res.begin());
    }
    return res;
}

// 减法运算
vector<int> subtract(vector<int>& a, vector<int>& b) {
    int n = max(a.size(), b.size());
    vector<int> res(n, 0);
    // 将两个数组逐位相减
    for (int i = 0; i < n; i++) {
        if (i < a.size()) res[i] += a[i];
        if (i < b.size()) res[i] -= b[i];
    }
    reverse(res.begin(), res.end());
    // 去除前导零
    while (res.size() > 1 && res[0] == 0) {
        res.erase(res.begin());
    }
    return res;
}

// 乘法运算
vector<int> multiply(vector<int>& a, vector<int>& b) {
    int n = a.size() + b.size() - 1;
    vector<int> res(n, 0);
    // 双重循环模拟多项式乘法
    for (int i = 0; i < a.size(); i++) {
        for (int j = 0; j < b.size(); j++) {
            res[i + j] += a[i] * b[j];
        }
    }
    reverse(res.begin(), res.end());
    // 去除前导零
    while (res.size() > 1 && res[0] == 0) {
        res.erase(res.begin());
    }
    return res;
}

int main() {
    string s;
    getline(cin, s);
    stringstream ss(s.substr(1, s.length() - 2));
    vector<int> a;
    int x;
    while (ss >> x) a.push_back(x);
    reverse(a.begin(), a.end());

    getline(cin, s);
    ss.clear();
    ss.str(s.substr(1, s.length() - 2));
    vector<int> b;
    while (ss >> x) b.push_back(x);
    reverse(b.begin(), b.end());

    char op;
    cin >> op;

    vector<int> res;
    if (op == '+') res = add(a, b);
    else if (op == '-') res = subtract(a, b);
    else if (op == '*') res = multiply(a, b);

    // 输出结果并格式化为多项式的数组表示
    cout << "[";
    for (int i = 0; i < res.size(); i++) {
        if (i != 0) cout << " ";
        cout << res[i];
    }
    cout << "]" << endl;

    return 0;
}
```

## BFS

### *[关灯](https://codefun2000.com/p/P2302)

> 小明需从屋子内标记为 “s” 的位置出发，通过上下左右移动到达标记为 “t” 的电源处，初始有一定水深且水面每单位时间上涨 1，小明每移动一步耗时 1 单位时间，仅能移动到未被水淹的方格（箱子方格需高度大于当前水位，0 方格必被淹，s 和 t 始终安全），若无法找到此类路线，小明需待在原地，最终需输出路径长度及各步坐标。

```c++
#include <bits/stdc++.h>
using namespace std;

typedef pair<int,int> P;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int H;
    cin >> H;

    int n, m;
    cin >> n >> m;

    vector<vector<string>> grid(n, vector<string>(m));
    vector<vector<int>> height(n, vector<int>(m, 0));

    P start, target;

    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            cin >> grid[i][j];
            if(grid[i][j] == "s") {
                start = {i, j};
                height[i][j] = 1e9; // 永不淹
            }
            else if(grid[i][j] == "t") {
                target = {i, j};
                height[i][j] = 1e9;
            }
            else {
                int h = stoi(grid[i][j]);
                height[i][j] = h;
            }
        }
    }

    // 最晚进入时间
    vector<vector<int>> deadline(n, vector<int>(m));
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            deadline[i][j] = height[i][j] - H;
        }
    }

    // BFS
    queue<P> q;
    vector<vector<int>> dist(n, vector<int>(m, -1));
    vector<vector<P>> pre(n, vector<P>(m, {-1,-1}));

    q.push(start);
    dist[start.first][start.second] = 0;

    int dx[4] = {1, -1, 0, 0};
    int dy[4] = {0, 0, 1, -1};

    while(!q.empty()) {
        auto [x, y] = q.front(); q.pop();

        for(int k = 0; k < 4; k++) {
            int nx = x + dx[k];
            int ny = y + dy[k];

            if(nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
            if(dist[nx][ny] != -1) continue;

            int nt = dist[x][y] + 1;

            // 必须在淹没前进入
            if(nt >= deadline[nx][ny]) continue;

            dist[nx][ny] = nt;
            pre[nx][ny] = {x, y};
            q.push({nx, ny});
        }
    }

    // 无法到达
    if(dist[target.first][target.second] == -1) {
        cout << 1 << "\n";
        cout << start.first << " " << start.second << "\n";
        return 0;
    }

    // 还原路径
    vector<P> path;
    P cur = target;
    while(cur != make_pair(-1, -1)) {
        path.push_back(cur);
        cur = pre[cur.first][cur.second];
    }

    reverse(path.begin(), path.end());

    cout << path.size() << "\n";
    for(auto &p : path) {
        cout << p.first << " " << p.second << "\n";
    }

    return 0;
}
```
